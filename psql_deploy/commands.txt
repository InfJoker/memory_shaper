CREATE TABLE auth_users (
    login                 varchar(80) PRIMARY KEY,
    password              varchar(65) NOT NULL,
    salt                  varchar(80) NOT NULL
);

CREATE TABLE users (
    login                 varchar(80) REFERENCES auth_users (login) ON DELETE CASCADE,
    first_name            varchar(80),
    second_name           varchar(80),
    nickname              varchar(80) PRIMARY KEY,
    age                   int,
    date_of_birth         date
);

CREATE TABLE decks (
    id                    int PRIMARY KEY,
    name                  varchar(80),
    thematic              varchar(80),
    total_card_num        int,
    description           text
);

CREATE TABLE cards (
    id                    int PRIMARY KEY,
    deck_id               int REFERENCES decks (id) ON DELETE CASCADE,
    in_deck_num           int,
    name                  varchar(80),
    translation           varchar(80),
    description           text
);

CREATE TABLE user_decks (
    id                    int PRIMARY KEY,
    user_nickname         varchar(80) REFERENCES users (nickname),
    deck_id               int REFERENCES decks (id),
    description           text
);

CREATE TABLE user_show_cards (
    id                    int PRIMARY KEY,
    user_nickname         varchar(80) REFERENCES users (nickname),
    card_id               int REFERENCES cards (id),
    next_show_date        date,
    description           text
);
